create database hex_maverick_75572_db;
use hex_maverick_75572_db;

CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Manager` (
  `Manager_ID` VARCHAR(45) NOT NULL,
  `Name` VARCHAR(45) NULL,
  `Username` VARCHAR(45) NULL,
  `Password` VARCHAR(45) NULL,
  PRIMARY KEY (`Manager_ID`));

CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Employee` (
  `Employee_ID` VARCHAR(45) NOT NULL,
  `Name` VARCHAR(45) NULL,
  `Password` VARCHAR(45) NULL,
  `Curr_Points` INT NULL,
  `Total_Ponts` INT NULL,
  PRIMARY KEY (`Employee_ID`));


CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Transaction` (
  `Transaction_ID` VARCHAR(45) NOT NULL,
  `NumOfItems` INT NOT NULL,
  `Point_Cost` INT NOT NULL,
  `Employee_Employee_ID` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`Transaction_ID`, `Employee_Employee_ID`),
  INDEX `fk_Transaction_Employee1_idx` (`Employee_Employee_ID` ASC) ,
  CONSTRAINT `fk_Transaction_Employee1`
    FOREIGN KEY (`Employee_Employee_ID`)
    REFERENCES `mydb`.`Employee` (`Employee_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);


CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Items` (
  `Item_ID` VARCHAR(45) NOT NULL,
  `Name` VARCHAR(45) NULL,
  `PtValue` INT NULL,
  PRIMARY KEY (`Item_ID`));



CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Employee_has_Items` (
  `Employee_Employee_ID` VARCHAR(45) NOT NULL,
  `Items_Item_ID` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`Employee_Employee_ID`, `Items_Item_ID`),
  INDEX `fk_Employee_has_Items_Items1_idx` (`Items_Item_ID` ASC) ,
  INDEX `fk_Employee_has_Items_Employee1_idx` (`Employee_Employee_ID` ASC) ,
  CONSTRAINT `fk_Employee_has_Items_Employee1`
    FOREIGN KEY (`Employee_Employee_ID`)
    REFERENCES `mydb`.`Employee` (`Employee_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `fk_Employee_has_Items_Items1`
    FOREIGN KEY (`Items_Item_ID`)
    REFERENCES `mydb`.`Items` (`Item_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Gives_Pts` (
  `Manager_Manager_ID` VARCHAR(45) NOT NULL,
  `Employee_Employee_ID` VARCHAR(45) NOT NULL,
  `Points` INT NULL);
