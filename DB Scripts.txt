CREATE SCHEMA IF NOT EXISTS `hex_maverick_75572_db`;
USE `hex_maverick_75572_db` ;

CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`employee` (
  `Employee_ID` INT NOT NULL AUTO_INCREMENT,
  `Username` VARCHAR(45) NULL DEFAULT NULL,
  `Password` VARCHAR(45) NULL DEFAULT NULL,
  `Curr_Points` INT NULL DEFAULT NULL,
  `Total_Ponts` INT NULL DEFAULT NULL,
  `name` VARCHAR(45) NULL DEFAULT NULL,
  PRIMARY KEY (`Employee_ID`));


CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`items` (
  `Item_ID` INT NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(45) NULL DEFAULT NULL,
  `PtValue` INT NULL DEFAULT NULL,
  PRIMARY KEY (`Item_ID`));


CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`manager` (
  `Manager_ID` INT NOT NULL AUTO_INCREMENT,
  `Name` VARCHAR(45) NULL DEFAULT NULL,
  `Username` VARCHAR(45) NULL DEFAULT NULL,
  `Password` VARCHAR(45) NULL DEFAULT NULL,
  PRIMARY KEY (`Manager_ID`));



CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`transaction` (
  `Transaction_ID` INT NOT NULL AUTO_INCREMENT,
  `NumOfItems` INT NOT NULL,
  `Point_Cost` INT NOT NULL,
  `Employee_ID` INT NOT NULL,
  PRIMARY KEY (`Transaction_ID`, `Employee_ID`))
ENGINE = InnoDB
AUTO_INCREMENT = 4
DEFAULT CHARACTER SET = utf8mb4
COLLATE = utf8mb4_0900_ai_ci;


CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Gives_Pts` (
  `Manager_ID` INT NOT NULL,
  `Employee_ID` INT NOT NULL,
  `NumOfPts` INT NULL,
  PRIMARY KEY (`Manager_ID`, `Employee_ID`),
  INDEX `Employee_ID_idx` (`Employee_ID` ASC),
  CONSTRAINT `Manager_ID`
    FOREIGN KEY (`Manager_ID`)
    REFERENCES `hex_maverick_75572_db`.`manager` (`Manager_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `Employee_ID`
    FOREIGN KEY (`Employee_ID`)
    REFERENCES `hex_maverick_75572_db`.`employee` (`Employee_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;


CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Makes` (
  `Employee_ID` INT NOT NULL,
  `Transaction_ID` INT NOT NULL,
  `Transaction_Date` DATE NULL,
  PRIMARY KEY (`Employee_ID`, `Transaction_ID`),
  INDEX `Transaction_ID_idx` (`Transaction_ID` ASC),
    FOREIGN KEY (`Employee_ID`)
    REFERENCES `hex_maverick_75572_db`.`employee` (`Employee_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
    FOREIGN KEY (`Transaction_ID`)
    REFERENCES `hex_maverick_75572_db`.`transaction` (`Transaction_ID`);

CREATE TABLE IF NOT EXISTS `hex_maverick_75572_db`.`Contains` (
  `Transaction_ID` INT NOT NULL,
  `Item_ID` INT NOT NULL,
  `Quantity` INT NULL,
  PRIMARY KEY (`Transaction_ID`, `Item_ID`),
  INDEX `Item_ID_idx` (`Item_ID` ASC),
  CONSTRAINT `Transaction_ID`
    FOREIGN KEY (`Transaction_ID`)
    REFERENCES `hex_maverick_75572_db`.`transaction` (`Transaction_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `Item_ID`
    FOREIGN KEY (`Item_ID`)
    REFERENCES `hex_maverick_75572_db`.`items` (`Item_ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
